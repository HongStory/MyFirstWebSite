SQL Language

  • cmd => bin ./mysql -uroot -p
  • create => CREATE DATABASE name;
  • drop => DROP DATABASE name;
  • show => SHOW DATABASES;
  • use => USE database name;

    • CREATE

  • create table =>
    CREATE TABLE name( id INT(11) NOT NULL
    AUTO_INCREMENT,
    title VARCHAR(100) NOT NULL,
    description TEXT NULL,
    created DATETIME NOT NULL,
    author VARCHAR(30) NULL,
    profile VARCHAR(100) NULL,
    PRIMARY KEY(id)
    );
  • DESC table name;
  • INSERT INTO table name
    (title,description,created,
    author,profile)
    VALUES('MySQL','MySQL is',NOW(),
    'user name','user profile');

    • READ

  • SELECT * FROM table name;
  • SELECT id,title,created FROM table name;
  • SELECT id,title FROM table name;
  • SELECT "author name";
  • SELECT * FROM table name
    WHERE author = 'name';
  • SELECT * FROM table name
    WHERE author = 'name'
    ORDER BY id DESC;
  • SELECT * FROM table name
    WHERE author = 'name'
    ORDER BY id DESC LIMIT 2;

    • UPDATE

  • Very important "WHERE" !!
  • UPDATE table name SET description=
    'new description', title='new title' WHERE id=2;

    • Delete

  • Very important "WHERE" !!
  • DELETE FROM table name WHERE id = 5;

    • Relational Database

  • SELECT * FROM table name1 LEFT JOIN
    table name2 ON table name1.author_id=table name2.id;
  • SELECT table name1.id,title,description,
    created,name,profile FROM table name1 LEFT JOIN table name2 ON
    table name1.author_id=
    table name2.id;
  • SELECT table name1.id AS table name1_id,
    title,description,created,name,
    profile FROM table name1
    LEFT JOIN table name2
    ON table name1.author_id=table name2.id;
